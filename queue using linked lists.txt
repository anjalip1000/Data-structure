#include <stdio.h>
#include <stdlib.h>

struct node {
	int data;
	struct node* next;
};
struct node* front;
struct node* rear;
void insert(struct node *ptr, int item) {
ptr = (struct node *) malloc (sizeof(struct node));
	if (ptr == NULL) {
		printf("\n<<<<<------OVERFLOW-------->>>>>");
		} else {
		ptr -> data = item;
		if (front == NULL) {
			front = ptr;
			rear = ptr;
			front -> next = NULL;
			rear -> next = NULL;
		} else {
			rear -> next = ptr;
			rear = ptr;
			rear->next = NULL;
	}}}
